  let _sharedClient: ReturnType<typeof createClient> | null = null;                                                                                                                                                                             
                                                                                                                                                                                                                                                
  export function setSharedClient(client: ReturnType<typeof createClient> | null) {

  }                                                                                                                                                                                                                                             
                                                                                                                                                                                                                                                
  export function getXingjingClient(): ReturnType<typeof createClient> {

  }
